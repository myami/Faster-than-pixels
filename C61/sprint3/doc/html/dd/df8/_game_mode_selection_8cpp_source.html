153void GameModeSelection::S_Input_Mouse(sf::Event event)
154{
155 if (event.type == sf::Event::MouseButtonPressed) {
156 if (Easy.IsSpriteClicked(_SceneManager->_GameManager->Windows)) {
157 DifficultySelection.setPosition(_SceneManager->_GameManager->View.getCenter());
158 DifficultySelection.move(sf::Vector2(-225.f, -215.f));
159 CurrentDifficulty = 0;
160 }
161 if (Medium.IsSpriteClicked(_SceneManager->_GameManager->Windows)) {
162 DifficultySelection.setPosition(_SceneManager->_GameManager->View.getCenter());
163 CurrentDifficulty = 1;
164 DifficultySelection.move(sf::Vector2(0.f, -215.f));
165 }
166 if (Hard.IsSpriteClicked(_SceneManager->_GameManager->Windows)) {
167 CurrentDifficulty = 2;

169 DifficultySelection.move(sf::Vector2(225.f, -215.f));

172 _SceneManager->ChangeScene("MainMenu");

174 if (Training.IsSpriteClicked(_SceneManager->_GameManager->Windows)) {
175 ModesSelection.setPosition(_SceneManager->_GameManager->View.getCenter());
176 ModesSelection.move(sf::Vector2(-265.f, 160.f));
177 CurrentModes = 0;
178 }
179 if (FastAction.IsSpriteClicked(_SceneManager->_GameManager->Windows)) {
180 CurrentModes = 1;

182 ModesSelection.move(sf::Vector2(10.f, 160.f));
183 }
184 if (StartGame.IsSpriteClicked(_SceneManager->_GameManager->Windows)) {
185 _SceneManager->ChangeScene("Game");
186 }
187
188 if (Seed.IsInputClicked(_SceneManager->_GameManager->Windows)) {
189 Seed.SetSelected(true);
190 }
191 if (Seed.isSelected && !Seed.IsInputClicked(_SceneManager->_GameManager->Windows)) {
192 Seed.SetSelected(false);
