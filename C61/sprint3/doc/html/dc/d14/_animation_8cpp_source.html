29 if (spritesToAnimate[currentFrame].getTexture()->getSize().x % numSprites.x != 0 or spritesToAnimate[currentFrame].getTexture()->getSize().y % numSprites.y != 0) {
30 //If the width and height of the sprite sheet aren't perfectly disivible the number of sprites, cast the result
